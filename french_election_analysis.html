def test_mention(df,ref_candidate,dict_nomention):
    for candidate in dict_nomention:
        if inlist(candidate,ref_candidate):
            df=df.loc[df["mention_"+candidate]==1].copy()
        else:
            df=df.loc[df["mention_"+candidate]==0].copy()
    return df
